```
---
agents:
  - name: series
    type: smart_title
    endpoint: http://weburg.net/series/all/?clever_title=1&template=0&last=0&sorts=date_update
    interval: 1m
    channel:
      type: channel
      username: weburg_series
    cache_size: 2
    print_categories: true
    skip_categories:
      - Сериал
```
